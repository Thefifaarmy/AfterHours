    function loadState() {
      const namesSaved = localStorage.getItem(`${APP_ID}:names`);
      const relSaved = localStorage.getItem(`${APP_ID}:relationship`);
      const settingsSaved = localStorage.getItem(`${APP_ID}:settings`);
      const favSaved = localStorage.getItem(`${APP_ID}:favorites`);

      if (namesSaved) state.names = JSON.parse(namesSaved);
      if (relSaved) state.relationship = JSON.parse(relSaved);
      if (settingsSaved) state.settings = JSON.parse(settingsSaved);
      if (favSaved) state.favorites = JSON.parse(favSaved);

      // Route screen based on setup completion
      if (state.names.p1 && state.names.p2) {
        state.currentScreen = 'dashboard';
      } else {
        state.currentScreen = 'onboarding';
      }
    }

    function saveState(key, data) {
      localStorage.setItem(`${APP_ID}:${key}`, JSON.stringify(data));
    }

    function showToast(msg) {
      const toast = document.getElementById('toast');
      const toastMsg = document.getElementById('toast-message');
      toastMsg.textContent = msg;
      
      toast.classList.remove('translate-y-[-100px]', 'opacity-0', 'pointer-events-none');
      toast.classList.add('translate-y-0', 'opacity-100');
      
      setTimeout(() => {
        toast.classList.add('translate-y-[-100px]', 'opacity-0', 'pointer-events-none');
        toast.classList.remove('translate-y-0', 'opacity-100');

    }

    function navigateTo(screenId) {
      state.currentScreen = screenId;
      document.querySelectorAll('.screen').forEach(scr => scr.classList.add('hidden'));
      
      const targetScreen = document.getElementById(`screen-${screenId}`);
      if (targetScreen) targetScreen.classList.remove('hidden');

      // Toggle Header visibility logic
      const headerControls = document.getElementById('header-controls');
      if (screenId === 'onboarding') {
        headerControls.classList.add('hidden');
        // Populate inputs from state if any
        document.getElementById('input-p1').value = state.names.p1;
        document.getElementById('input-p2').value = state.names.p2;
        document.getElementById('input-anniversary').value = state.relationship.anniversary;
      } else {
        headerControls.classList.remove('hidden');
      }

      if (screenId === 'dashboard') {
        updateDashboardData();
      }

      if (screenId === 'favorites') {
        renderFavorites();
      }

      if (screenId === 'settings') {
        syncSettingsUI();
      }

      triggerHaptic();
      window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    // --- DYNAMIC QUESTION PROMPT GENERATION ENGINE ---
    function generatePrompt(categoryKey) {
      const catData = ENGINE_DATABASE[categoryKey] || ENGINE_DATABASE['relationship'];
      const isTemplate = Math.random() > 0.45 && catData.templates.length > 0;
      let rawText = "";
      let target = "both";

      if (isTemplate) {
        const template = catData.templates[Math.floor(Math.random() * catData.templates.length)];
        let processed = template;
        
        processed = processed.replace(/{p1}/g, state.names.p1 || "Player 1");
        processed = processed.replace(/{p2}/g, state.names.p2 || "Player 2");
        
        if (processed.includes("{other}")) {
          const turn = Math.random() > 0.5 ? 'p1' : 'p2';
          processed = processed.replace(/{other}/g, turn === 'p1' ? state.names.p1 : state.names.p2);
          target = turn === 'p1' ? 'p2' : 'p1';
        }

        Object.keys(TEMPLATE_VARIABLES).forEach(vKey => {
          const regex = new RegExp(`{${vKey}}`, 'g');
          if (processed.match(regex)) {
            const varChoices = TEMPLATE_VARIABLES[vKey];
            const choice = varChoices[Math.floor(Math.random() * varChoices.length)];
            processed = processed.replace(regex, choice);
          }
        });

        rawText = processed;
      } else {
        const item = catData.static[Math.floor(Math.random() * catData.static.length)];
        rawText = item.question;
        target = item.target;
      }

      let targetLabel = "Question for Both";
      if (target === 'p1') {
        targetLabel = `Question for ${state.names.p1}`;
      } else if (target === 'p2') {
        targetLabel = `Question for ${state.names.p2}`;
      } else if (target === 'male' || target === 'female') {
        const assign = Math.random() > 0.5 ? state.names.p1 : state.names.p2;
        targetLabel = `Question for ${assign}`;
      }

      return {
        id: `${categoryKey}-${Date.now()}-${Math.floor(Math.random() * 1000)}`,
        category: categoryKey,
        question: rawText,
        targetLabel: targetLabel,
        difficulty: 'Medium'
      };
    }

    // --- GAMEPLAY CORE CONTROLLER ---
    function startCategoryDeck(catId) {
      triggerHaptic();
      const cat = CATEGORIES.find(c => c.id === catId);
      state.currentCategory = cat;
      
      const firstPrompt = generatePrompt(catId);
      state.deckHistory = [firstPrompt];
      state.deckHistoryIndex = 0;
      state.currentQuestion = firstPrompt;
      
      document.getElementById('btn-spin-again').classList.add('hidden');
      updateGameplayUI();
      navigateTo('game');
    }

    function startRoulette() {
      triggerHaptic();
      const randomCat = CATEGORIES[Math.floor(Math.random() * CATEGORIES.length)];
      state.currentCategory = randomCat;
      
      const prompt = generatePrompt(randomCat.id);
      state.currentQuestion = prompt;
      state.deckHistory = [prompt];
      state.deckHistoryIndex = 0;

      document.getElementById('btn-spin-again').classList.remove('hidden');
      updateGameplayUI();
      navigateTo('game');
    }

      card.className = "relative min-h-[42vh] rounded-3xl border flex flex-col justify-between p-8 shadow-2xl transition-all duration-700";
      
      if (cat.isMidnight) {
        card.classList.add('from-indigo-950', 'via-purple-900', 'to-slate-950', 'border-purple-500/30', 'text-white', 'bg-gradient-to-br');
      } else if (cat.isAdult) {
        card.classList.add('from-stone-950', 'via-rose-950', 'to-stone-900', 'border-red-500/30', 'text-white', 'bg-gradient-to-br');
      } else {
        card.classList.add('from-white/10', 'to-white/5', 'backdrop-blur-xl', 'border-white/10', 'bg-gradient-to-br');
      }

      // Update Prompt and Target Banner Text
      document.getElementById('card-target-label').textContent = q.targetLabel;
      document.getElementById('card-question-text').textContent = `"${q.question}"`;

      // Update Favorite Icon
      const starIcon = document.getElementById('card-fav-star');
      const isFav = state.favorites.some(f => f.question === q.question);
      if (isFav) {
        starIcon.className = "w-6 h-6 transition-all text-amber-400 fill-amber-400 scale-110";
      } else {
        starIcon.className = "w-6 h-6 transition-all opacity-40 hover:opacity-100 text-white";
      }

      // Update Prev Button disabled status
      const prevBtn = document.getElementById('btn-prev-card');
      if (state.deckHistoryIndex <= 0) {
        prevBtn.classList.add('opacity-20', 'cursor-not-allowed');
        prevBtn.setAttribute('disabled', 'true');
      } else {
        prevBtn.classList.remove('opacity-20', 'cursor-not-allowed');
        prevBtn.removeAttribute('disabled');
      }

      applyFontSize();
      lucide.createIcons();
    }

    function nextQuestion() {
      triggerHaptic();
      if (state.deckHistoryIndex < state.deckHistory.length - 1) {
        state.deckHistoryIndex++;
        state.currentQuestion = state.deckHistory[state.deckHistoryIndex];
      } else {
        const newPrompt = generatePrompt(state.currentCategory.id);
        state.deckHistory.push(newPrompt);
        state.deckHistoryIndex = state.deckHistory.length - 1;
        state.currentQuestion = newPrompt;
      }
      updateGameplayUI();
    }

    function prevQuestion() {
      if (state.deckHistoryIndex > 0) {
        triggerHaptic();
        state.deckHistoryIndex--;
        state.currentQuestion = state.deckHistory[state.deckHistoryIndex];
        updateGameplayUI();
      }
    }

    function shuffleQuestion() {
      triggerHaptic();
      const fresh = generatePrompt(state.currentCategory.id);
      state.currentQuestion = fresh;
      state.deckHistory.push(fresh);
      state.deckHistoryIndex = state.deckHistory.length - 1;
      updateGameplayUI();
    }

    function toggleCurrentFavorite() {
      triggerHaptic();
      const q = state.currentQuestion;
      const isFav = state.favorites.some(f => f.question === q.question);
      
      if (isFav) {
        state.favorites = state.favorites.filter(f => f.question !== q.question);
        showToast('Removed from Favourites');
      } else {
        const newFav = {
          ...q,
          savedAt: new Date().toLocaleDateString(),
          categoryTitle: CATEGORIES.find(c => c.id === q.category)?.title || 'Custom'
        };
        state.favorites.unshift(newFav);
        showToast('Added to Favourites! ❤️');
      }
      
      saveState('favorites', state.favorites);
      updateGameplayUI();
    }

    // --- FAVORITES SCREEN INTERACTIONS ---
    function renderFavorites(query = '') {
      const list = document.getElementById('favorites-list');
      const filtered = state.favorites.filter(f => f.question.toLowerCase().includes(query.toLowerCase()));
      
      document.getElementById('favs-title').textContent = `Saved Deck (${state.favorites.length})`;

      if (filtered.length === 0) {
        list.innerHTML = `
          <div class="text-center py-12 bg-white/5 border border-white/5 rounded-2xl">
            <i data-lucide="bookmark" class="w-8 h-8 mx-auto mb-2 opacity-30 text-pink-400"></i>
            <p class="text-sm opacity-60">No saved cards match your search.</p>
          </div>
        `;
      } else {
        list.innerHTML = filtered.map(item => `
          <div class="p-4 bg-white/5 border border-white/5 hover:border-white/10 rounded-2xl flex items-start justify-between gap-3 transition-all">
            <div class="flex-1">
              <span class="text-[9px] bg-white/10 text-white/75 px-2 py-0.5 rounded-full font-bold uppercase tracking-wider mb-2 inline-block">
                ${item.categoryTitle}
              </span>
              <p class="text-sm font-medium tracking-wide leading-relaxed">
                "${item.question}"
              </p>
            </div>
            <div class="flex flex-col gap-1 items-end">
              <button onclick="playSavedCard('${item.id}')" class="bg-pink-500/20 text-pink-300 text-[10px] font-bold px-2.5 py-1 rounded-lg hover:bg-pink-500/30 transition-all">
                Play Card
              </button>
              <button onclick="removeFavoriteCard('${item.id}')" class="p-1.5 rounded-lg hover:bg-white/10 text-rose-500 transition-all">
                <i data-lucide="trash-2" class="w-4 h-4"></i>
              </button>
            </div>
          </div>
        `).join('');
      }
      lucide.createIcons();
    }

    function filterFavorites(value) {
      renderFavorites(value);
    }

    function playSavedCard(cardId) {
      triggerHaptic();
      const card = state.favorites.find(f => f.id === cardId);
      if (card) {
        state.currentCategory = CATEGORIES.find(c => c.id === card.category) || CATEGORIES[0];
        state.currentQuestion = card;
        state.deckHistory = [card];
        state.deckHistoryIndex = 0;
        
        document.getElementById('btn-spin-again').classList.add('hidden');
        updateGameplayUI();
        navigateTo('game');
      }
    }

    function removeFavoriteCard(cardId) {
      triggerHaptic();
      state.favorites = state.favorites.filter(f => f.id !== cardId);
      saveState('favorites', state.favorites);
      renderFavorites(document.getElementById('favs-search-input').value);
      showToast('Removed from Favourites');
    }

    // --- SETTINGS CONTROLLER ---
    function syncSettingsUI() {
      // Font Buttons
      document.querySelectorAll('[id^="font-btn-"]').forEach(btn => {
        btn.className = "py-1.5 text-xs font-bold rounded-lg uppercase tracking-wider transition-all hover:bg-white/5 opacity-60";
      });
      const activeFontBtn = document.getElementById(`font-btn-${state.settings.fontSize}`);
      if (activeFontBtn) {
        activeFontBtn.className = "py-1.5 text-xs font-bold rounded-lg uppercase tracking-wider transition-all bg-gradient-to-r from-pink-500 to-rose-500 text-white";
      }

      // Switches
      updateSwitchUI('haptics');
      updateSwitchUI('animations');
      updateSwitchUI('darkMode');
    }

    function updateSwitchUI(key) {
      const toggle = document.getElementById(`toggle-${key}`);
      const dot = toggle.querySelector('div');
      
      if (state.settings[key]) {
        toggle.className = "w-12 h-6 rounded-full p-1 transition-all bg-pink-500";
        dot.className = "w-4 h-4 rounded-full bg-white transition-all transform translate-x-6";
      } else {
        toggle.className = "w-12 h-6 rounded-full p-1 transition-all bg-white/10";
        dot.className = "w-4 h-4 rounded-full bg-white transition-all transform translate-x-0";
      }
    }

    function setFontSize(size) {
      triggerHaptic();
      state.settings.fontSize = size;
      saveState('settings', state.settings);
      syncSettingsUI();
    }

    function applyFontSize() {
      const txt = document.getElementById('card-question-text');
      txt.className = "font-medium tracking-tight leading-relaxed text-center w-full";
      
      switch (state.settings.fontSize) {
        case 'sm': txt.classList.add('text-xl', 'md:text-2xl'); break;
        case 'md': txt.classList.add('text-2xl', 'md:text-3xl'); break;
        case 'lg': txt.classList.add('text-3xl', 'md:text-4xl'); break;
        case 'xl': txt.classList.add('text-4xl', 'md:text-5xl'); break;
      }
    }

    function toggleSetting(key) {
      triggerHaptic();
      state.settings[key] = !state.settings[key];
      saveState('settings', state.settings);
      syncSettingsUI();
      applyThemeSettings();
    }

    function applyThemeSettings() {
      // Update body background lock state
      const body = document.body;
      const themeIcon = document.getElementById('theme-icon');
      
      if (state.settings.darkMode) {
        body.className = "min-h-screen relative flex flex-col justify-between overflow-x-hidden select-none transition-colors duration-500 bg-slate-950 text-white";
        themeIcon.setAttribute('data-lucide', 'sun');
        themeIcon.className.baseVal = "w-4 h-4 text-amber-400";
      } else {
        body.className = "min-h-screen relative flex flex-col justify-between overflow-x-hidden select-none transition-colors duration-500 bg-rose-50 text-slate-900";
        themeIcon.setAttribute('data-lucide', 'moon');
        themeIcon.className.baseVal = "w-4 h-4 text-indigo-700";
      }
      lucide.createIcons();
    }

    function toggleTheme() {
      toggleSetting('darkMode');
    }

    function confirmResetAll() {
      triggerHaptic();
      if (confirm("Are you absolutely sure you want to clear all your saved settings and favourites?")) {
        localStorage.clear();
        state = {
          names: { p1: '', p2: '' },
          relationship: { status: 'Dating', anniversary: '' },
          settings: { haptics: true, animations: true, darkMode: true, fontSize: 'lg' },
          favorites: [],
          currentScreen: 'onboarding',
          currentCategory: null,
          currentQuestion: null,
          deckHistory: [],
          deckHistoryIndex: -1
        };
        initUI();
        showToast('All data successfully cleared');
      }
    }

    // --- VISUAL AMBIENT PARTICLE ENGINE ---
    function initFloatingHearts() {
      const container = document.getElementById('heart-container');
      
      setInterval(() => {
        if (!state.settings.animations) return;
        
        const heart = document.createElement('div');
        heart.className = 'heart-particle';
        heart.textContent = '♥';
        
        const xPos = Math.random() * 100;
        const size = Math.random() * 15 + 10;
        const duration = Math.random() * 6 + 4;
        const delay = Math.random() * 2;
        
        heart.style.left = `${xPos}%`;
        heart.style.fontSize = `${size}px`;
        heart.style.animation = `floatUp ${duration}s linear infinite`;
        heart.style.animationDelay = `${delay}s`;
        
        container.appendChild(heart);
        
        // Cleanup old hearts
        setTimeout(() => {
          heart.remove();
        }, (duration + delay) * 1000);
      }, 2000);
    }
  </script>
</body>
</html>
